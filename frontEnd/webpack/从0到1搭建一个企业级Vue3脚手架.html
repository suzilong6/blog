# 处理样式文件（css, less, sass）

npm install --save-dev style-loader css-loader
npm install --save-dev less-loader less
npm install --save-dev sass-loader node-sass
1
2
3
module.exports = {
  module: {
    rules: [
		 {
        test: /\.css$/,
        // 注意：loader执行顺序：从后到前
        use: [
          "style-loader", // 将js中css通过style标签加入到html中
          "css-loader", // 将css资源编译成commonJS的模块到js中
        ],
      },
      {
        test: /\.less$/,
        use: [
          "style-loader",
          "css-loader",
          "less-loader", // 将less编译成css
        ],
      },
      {
        test: /\.scss$/,
        use: [
          "style-loader",
          "css-loader",
          "sass-loader", // 将Sass编译成CSS
        ],
      },
    ]
  },
};

# 处理图片等静态资源

在 webpack 5 之前，通常使用：

  • raw-loader (opens new window) 将文件导入为字符串
  • url-loader (opens new window) 将文件作为 data URI 内联到 bundle 中
  • file-loader (opens new window) 将文件发送到输出目录

资源模块类型(asset module type)，通过添加 4 种新的模块类型，来替换所有这些 loader：

  • asset/resource 发送一个单独的文件并导出 URL。之前通过使用 file-loader 实现。
  • asset/inline 导出一个资源的 data URI。之前通过使用 url-loader 实现。
  • asset/source 导出资源的源代码。之前通过使用 raw-loader 实现。
  • asset 在导出一个 data URI 和发送一个单独的文件之间自动选择。之前通过使用 url-loader，并且配置资源体积限制实现。
module.exports = {
	module: {
		rules: [
       {
         test: /\.(png|svg|jpg|jpeg|gif)$/,
         type: 'asset/resource'
       },
		]
	}
};

# 集成 TypeScript

vue3 源码已经全部用ts重写了，因此 vue3 对于ts的支持非常nice，下面我们将ts集成到我们的项目。

npm install --save-dev typescript ts-loader
1
module.exports = {
  module: {
    rules: [
      {
        test: /\.tsx?$/,
        exclude: /node_modules/,
        use: 'ts-loader',
      },
    ],
  },
  // 配置模块如何解析。
  resolve: {
    // 尝试按顺序解析这些后缀名。如果有多个文件有相同的名字，但后缀名不同，webpack 会解析列在数组首位的后缀的文件 并跳过其余的后缀。
    extensions: ['.ts', '.js', '.json'],
  },
};

# 使用 webpack-dev-server

在每次编译代码时，手动运行 npm run build 会显得很麻烦。

webpack 提供几种可选方式，帮助你在代码发生变化后自动编译代码：

  1. webpack's Watch Mode (opens new window)
  2. webpack-dev-server (opens new window)
  3. webpack-dev-middleware (opens new window)

而webpack-dev-server适合大部分场景，因此我们将其集成到我们的环境中

npm install --save-dev webpack-dev-server
1

package.json



 



{
  "scripts": {
    "start": "webpack serve --open"
  },
}
1
2
3
4
5

现在运行 npm run start 就可以开启一个开发服务器，修改代码也会自动编译。

# 集成Router 和 Vuex

npm install vue-router@4 -S
npm install vuex@next -S
1
2

创建router/index.ts文件

import { createRouter, createWebHashHistory, RouteRecordRaw } from 'vue-router';
import Home from '../views/Home.vue';
import About from '../views/About.vue';

const routes: Array<RouteRecordRaw> = [
  {
    path: '/',
    name: 'home',
    component: Home,
  },
  {
    path: '/about',
    name: 'about',
    component: About,
  },
];

const router = createRouter({
  history: createWebHashHistory('/#/'),
  routes,
});

export default router;
